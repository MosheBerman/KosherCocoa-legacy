028 * 
029 * This open source Java code was originally ported by <a href="http://www.facebook.com/avromf">Avrom Finkelstien</a>

032 * 
033 * The methods used to obtain the parsha were derived from the source code of <a
034 * href="http://www.sadinoff.com/hebcal/">HebCal</a> by Danny Sadinoff and JCal for the Mac by Frank Yellin. Both based
035 * their code on routines by Nachum Dershowitz and Edward M. Reingold. The class allows setting whether the parsha and
036 * holiday scheme follows the Israel scheme or outside Israel scheme. The default is the outside Israel scheme.

043 * <li>Haftorah (various minhagim)</li>
044 * <li>Daf Yomi Yerushalmi, Mishna yomis etc)</li>
045 * <li>Support showing the upcoming parsha for the middle of the week</li>
046 * </ol>
047 * 
048 * @see java.util.Date
049 * @see java.util.Calendar
050 * @author &copy; Avrom Finkelstien 2002
051 * @author &copy; Eliyahu Hershfeld 2011

200        /**
201         * Returns an index of the Jewish holiday or fast day for the current day, or a null if there is no holiday for this
202         * day.
203         * 
204         * @return A String containing the holiday name or an empty string if it is not a holiday.
205         */
206        public int getYomTovIndex() {
207                // check by month (starts from Nissan)
208                switch (getJewishMonth()) {
209                case NISSAN:
210                        if (getJewishDayOfMonth() == 14) {
211                                return EREV_PESACH;
212                        } else if (getJewishDayOfMonth() == 15 || getJewishDayOfMonth() == 21
213                                        || (!inIsrael && (getJewishDayOfMonth() == 16 || getJewishDayOfMonth() == 22))) {
214                                return PESACH;
215                        } else if (getJewishDayOfMonth() >= 17 && getJewishDayOfMonth() <= 20
216                                        || (getJewishDayOfMonth() == 16 && inIsrael)) {
217                                return CHOL_HAMOED_PESACH;
218                        }
219                        if (isUseModernHolidays()
220                                        && ((getJewishDayOfMonth() == 26 && getDayOfWeek() == 5)
221                                                        || (getJewishDayOfMonth() == 28 && getDayOfWeek() == 1)
222                                                        || (getJewishDayOfMonth() == 27 && getDayOfWeek() == 3) || (getJewishDayOfMonth() == 27 && getDayOfWeek() == 5))) {
223                                return YOM_HASHOAH;
224                        }
225                        break;
226                case IYAR:
227                        if (isUseModernHolidays()
228                                        && ((getJewishDayOfMonth() == 4 && getDayOfWeek() == 3)
229                                                        || ((getJewishDayOfMonth() == 3 || getJewishDayOfMonth() == 2) && getDayOfWeek() == 4) || (getJewishDayOfMonth() == 5 && getDayOfWeek() == 2))) {
230                                return YOM_HAZIKARON;
231                        }
232                        // if 5 Iyar falls on Wed Yom Haatzmaut is that day. If it fal1s on Friday or Shabbos it is moved back to
233                        // Thursday. If it falls on Monday it is moved to Tuesday
234                        if (isUseModernHolidays()
235                                        && ((getJewishDayOfMonth() == 5 && getDayOfWeek() == 4)
236                                                        || ((getJewishDayOfMonth() == 4 || getJewishDayOfMonth() == 3) && getDayOfWeek() == 5) || (getJewishDayOfMonth() == 6 && getDayOfWeek() == 3))) {
237                                return YOM_HAATZMAUT;
238                        }
239                        if (getJewishDayOfMonth() == 14) {
240                                return PESACH_SHENI;
241                        }
242                        if (isUseModernHolidays() && getJewishDayOfMonth() == 28) {
243                                return YOM_YERUSHALAYIM;
244                        }
245                        break;
246                case SIVAN:
247                        if (getJewishDayOfMonth() == 5) {
248                                return EREV_SHAVUOS;
249                        } else if (getJewishDayOfMonth() == 6 || (getJewishDayOfMonth() == 7 && !inIsrael)) {
250                                return SHAVUOS;
251                        }
252                        break;
253                case TAMMUZ:
254                        // push off the fast day if it falls on Shabbos
255                        if ((getJewishDayOfMonth() == 17 && getDayOfWeek() != 7)
256                                        || (getJewishDayOfMonth() == 18 && getDayOfWeek() == 1)) {
257                                return SEVENTEEN_OF_TAMMUZ;
258                        }
259                        break;
260                case AV:
261                        // if Tisha B'av falls on Shabbos, push off until Sunday
262                        if ((getDayOfWeek() == 1 && getJewishDayOfMonth() == 10)
263                                        || (getDayOfWeek() != 7 && getJewishDayOfMonth() == 9)) {
264                                return TISHA_BEAV;
265                        } else if (getJewishDayOfMonth() == 15) {
266                                return TU_BEAV;
267                        }
268                        break;
269                case ELUL:
270                        if (getJewishDayOfMonth() == 29) {
271                                return EREV_ROSH_HASHANA;
272                        }
273                        break;
274                case TISHREI:
275                        if (getJewishDayOfMonth() == 1 || getJewishDayOfMonth() == 2) {
276                                return ROSH_HASHANA;
277                        } else if ((getJewishDayOfMonth() == 3 && getDayOfWeek() != 7)
278                                        || (getJewishDayOfMonth() == 4 && getDayOfWeek() == 1)) {
279                                // push off Tzom Gedalia if it falls on Shabbos
280                                return FAST_OF_GEDALYAH;
281                        } else if (getJewishDayOfMonth() == 9) {
282                                return EREV_YOM_KIPPUR;
283                        } else if (getJewishDayOfMonth() == 10) {
284                                return YOM_KIPPUR;
285                        } else if (getJewishDayOfMonth() == 14) {
286                                return EREV_SUCCOS;
287                        }
288                        if (getJewishDayOfMonth() == 15 || (getJewishDayOfMonth() == 16 && !inIsrael)) {
289                                return SUCCOS;
290                        }
291                        if (getJewishDayOfMonth() >= 17 && getJewishDayOfMonth() <= 20 || (getJewishDayOfMonth() == 16 && inIsrael)) {
292                                return CHOL_HAMOED_SUCCOS;
293                        }
294                        if (getJewishDayOfMonth() == 21) {
295                                return HOSHANA_RABBA;
296                        }
297                        if (getJewishDayOfMonth() == 22) {
298                                return SHEMINI_ATZERES;
299                        }
300                        if (getJewishDayOfMonth() == 23 && !inIsrael) {
301                                return SIMCHAS_TORAH;
302                        }
303                        break;
304                case KISLEV: // no yomtov in CHESHVAN
305                        // if (getJewishDayOfMonth() == 24) {
306                        // return EREV_CHANUKAH;
307                        // } else
308                        if (getJewishDayOfMonth() >= 25) {
309                                return CHANUKAH;
310                        }
311                        break;
312                case TEVES:
313                        if (getJewishDayOfMonth() == 1 || getJewishDayOfMonth() == 2
314                                        || (getJewishDayOfMonth() == 3 && isKislevShort())) {
315                                return CHANUKAH;
316                        } else if (getJewishDayOfMonth() == 10) {
317                                return TENTH_OF_TEVES;
318                        }
319                        break;
320                case SHEVAT:
321                        if (getJewishDayOfMonth() == 15) {
322                                return TU_BESHVAT;
323                        }
324                        break;
325                case ADAR:
326                        if (!isJewishLeapYear()) {
327                                // if 13th Adar falls on Friday or Shabbos, push back to Thursday
328                                if (((getJewishDayOfMonth() == 11 || getJewishDayOfMonth() == 12) && getDayOfWeek() == 5)
329                                                || (getJewishDayOfMonth() == 13 && !(getDayOfWeek() == 6 || getDayOfWeek() == 7))) {
330                                        return FAST_OF_ESTHER;
331                                }
332                                if (getJewishDayOfMonth() == 14) {
333                                        return PURIM;
334                                } else if (getJewishDayOfMonth() == 15) {
335                                        return SHUSHAN_PURIM;
336                                }
337                        } else { // else if a leap year
338                                if (getJewishDayOfMonth() == 14) {
339                                        return PURIM_KATAN;
340                                }
341                        }
342                        break;
343                case ADAR_II:
344                        // if 13th Adar falls on Friday or Shabbos, push back to Thursday
345                        if (((getJewishDayOfMonth() == 11 || getJewishDayOfMonth() == 12) && getDayOfWeek() == 5)
346                                        || (getJewishDayOfMonth() == 13 && !(getDayOfWeek() == 6 || getDayOfWeek() == 7))) {
347                                return FAST_OF_ESTHER;
348                        }
349                        if (getJewishDayOfMonth() == 14) {
350                                return PURIM;
351                        } else if (getJewishDayOfMonth() == 15) {
352                                return SHUSHAN_PURIM;
353                        }
354                        break;
355                }
356                // if we get to this stage, then there are no holidays for the given date return -1
357                return -1;
358        }

361         * Returns a String of the Jewish holiday or fast day for the current day, or a null if there is no holiday for this
362         * day. Has no "modern" holidays.

367                return new HebrewDateFormatter().formatYomTov(this);

370        /**
371         * Returns true if the current day is Yom Tov. The method returns false for Chanukah, Erev Yom tov and fast days.
372         * 
373         * @return true if the current day is a Yom Tov
374         * @see #isErevYomTov()
375         * @see #isTaanis()
376         */
377        public boolean isYomTov() {
378                int holidayIndex = getYomTovIndex();
379                if (isErevYomTov() || holidayIndex == CHANUKAH || (isTaanis() && holidayIndex != YOM_KIPPUR)) {
380                        return false;
381                }
382                return getYomTovIndex() != -1;
383        }
384
385        /**
386         * Returns true if the current day is erev Yom Tov. The method returns true for Erev - Pesach, Shavuos, Rosh
387         * Hashana, Yom Kippur and Succos.
388         * 
389         * @return true if the current day is Erev - Pesach, Shavuos, Rosh Hashana, Yom Kippur and Succos
390         * @see #isYomTov()
391         */
392        public boolean isErevYomTov() {
393                int holidayIndex = getYomTovIndex();
394                return holidayIndex == EREV_PESACH || holidayIndex == EREV_SHAVUOS || holidayIndex == EREV_ROSH_HASHANA
395                                || holidayIndex == EREV_YOM_KIPPUR || holidayIndex == EREV_SUCCOS;
396        }
397
398        /**
399         * Returns true if the current day is Erev Rosh Chodesh. Returns false for Erev Rosh Hashana
400         * 
401         * @return true if the current day is Erev Rosh Chodesh. Returns false for Erev Rosh Hashana
402         * @see #isRoshChodesh()
403         */
404        public boolean isErevRoshChodesh() {
405                // Erev Rosh Hashana is not Erev Rosh Chodesh.
406                return (getJewishDayOfMonth() == 29 && getJewishMonth() != ELUL);
407        }
408
409        /**
410         * Return true if the day is a Taanis (fast day). Return true for 17 of Tammuz, Tisha B'Av, Yom Kippur, Fast of
411         * Gedalyah, 10 of Teves and the Fast of Esther
412         * 
413         * @return true if today is a fast day
414         */
415        public boolean isTaanis() {
416                int holidayIndex = getYomTovIndex();
417                return holidayIndex == SEVENTEEN_OF_TAMMUZ || holidayIndex == TISHA_BEAV || holidayIndex == YOM_KIPPUR
418                                || holidayIndex == FAST_OF_GEDALYAH || holidayIndex == TENTH_OF_TEVES || holidayIndex == FAST_OF_ESTHER;
419        }
420
421        /**
422         * Returns the day of Chanukah or -1 if it is not Chanukah.
423         * 
424         * @return the day of Chanukah or -1 if it is not Chanukah.
425         */
426        public int getDayOfChanukah() {
427                if (isChanukah()) {
428                        if (getJewishMonth() == KISLEV) {
429                                return getJewishDayOfMonth() - 24;
430                        } else { // teves
431                                return isKislevShort() ? getJewishDayOfMonth() + 5 : getJewishDayOfMonth() + 4;
432                        }
433                } else {
434                        return -1;
435                }
436        }
437
438        public boolean isChanukah() {
439                return getYomTovIndex() == CHANUKAH;
440        }

556                // index= the index number of the parsha name

670         * Returns the int value of the Omer day or -1 if the day is not in the omer
671         * 
672         * @return The Omer count as an int or -1 if it is not a day of the Omer.
673         */
674        public int getDayOfOmer() {
675                int omer = -1; // not a day of the Omer
676
677                // if Nissan and second day of Pesach and on
678                if (getJewishMonth() == NISSAN && getJewishDayOfMonth() >= 16) {
679                        omer = getJewishDayOfMonth() - 15;
680                        // if Iyar
681                } else if (getJewishMonth() == IYAR) {
682                        omer = getJewishDayOfMonth() + 15;
683                        // if Sivan and before Shavuos
684                } else if (getJewishMonth() == SIVAN && getJewishDayOfMonth() < 6) {
685                        omer = getJewishDayOfMonth() + 44;
686                }
687                return omer;
688        }

690        /**
691         * Returns the molad in Standard Time in Yerushalayim as a Date. The traditional calculation uses local time. This
692         * method subtracts 20.94 minutes (20 minutes and 56.496 seconds) from the local time (Har Habayis with a longitude
693         * of 35.2354&deg; is 5.2354&deg; away from the %15 timezone longitude) to get to standard time. This method
694         * intentionally uses standard time and not dailight savings time. Java will implicitly format the time to the
695         * default (or set) Timezone.
696         * 
697         * @param jewishYear
698         *            the Jewish year.
699         * @param jewishMonth
700         *            the Jewish month. The method expects a 1 for Nissan ... 12 for Adar and 13 for Adar II. Use the
701         *            constants {@link #NISSAN} ... {@link #ADAR} (or {@link #ADAR_II} for a leap year Adar II) to avoid any
702         *            confusion.
703         * @return the Date representing the moment of the molad in Yerushalayim standard time (GMT + 2)
704         */
705        public static Date getMoladAsDate(int jewishYear, int jewishMonth) {
706                JewishDate molad = JewishDate.getMolad(jewishYear, jewishMonth);
707                String locationName = "Jerusalem, Israel";
708
709                double latitude = 31.778; // Har Habayis latitude
710                double longitude = 35.2354; // Har Habayis longitude
711
712                // The molad calculation always extepcst output in standard time. Using "Asia/Jerusalem" timezone will incorrect
713                // adjust for DST.
714                TimeZone yerushalayimStandardTZ = TimeZone.getTimeZone("GMT+2");
715                GeoLocation geo = new GeoLocation(locationName, latitude, longitude, yerushalayimStandardTZ);
716                Calendar cal = Calendar.getInstance(geo.getTimeZone());
717                cal.clear();
718                double moladSeconds = molad.getMoladChalakim() * 10 / 3;
719                cal.set(molad.getGregorianYear(), molad.getGregorianMonth(), molad.getGregorianDayOfMonth(),
720                                molad.getMoladHours(), molad.getMoladMinutes(), (int) moladSeconds);
721                cal.set(Calendar.MILLISECOND, (int) (1000 * (moladSeconds - (int) moladSeconds)));
722                // subtract local time difference of 20.94 minutes (20 minutes and 56.496 seconds) to get to Standard time
723                cal.add(Calendar.MILLISECOND, -1 * (int) geo.getLocalMeanTimeOffset());
724                return cal.getTime();
725        }
