017package net.sourceforge.zmanim.hebrewcalendar;
018
019import java.util.Calendar;
020import java.util.Date;
021import java.util.GregorianCalendar;
022
023/**
024 * This class calculates the Daf Yomi page (daf) for a given date. The class currently only supports Daf Yomi Bavli, but
025 * may cover Yerushalmi, Mishna Yomis etc in the future.
026 * 
027 * @author &copy; Bob Newell (original C code)
028 * @author &copy; Eliyahu Hershfeld 2011
029 * @version 0.0.1
030 */
031public class YomiCalculator {
032
033        private static Date dafYomiStartDate = new GregorianCalendar(1923, Calendar.SEPTEMBER, 11).getTime();
034        private static int dafYomiJulianStartDay = getJulianDay(dafYomiStartDate);
035        private static Date shekalimChangeDate = new GregorianCalendar(1975, Calendar.JUNE, 24).getTime();
036        private static int shekalimJulianChangeDay = getJulianDay(shekalimChangeDate);
037
038        /**
039         * Returns the <a href="http://en.wikipedia.org/wiki/Daf_yomi">Daf Yomi</a> <a
040         * href="http://en.wikipedia.org/wiki/Talmud">Bavli</a> {@link Daf} for a given date. The first Daf Yomi cycle
041         * started on Rosh Hashana 5684 (September 11, 1923) and calculations prior to this date will result in an
042         * IllegalArgumentException thrown. For historical calculations (supported by this method), it is important to note
043         * that a change in length of the cycle was instituted starting in the eighth Daf Yomi cycle beginning on June 24,
044         * 1975. The Daf Yomi Bavli cycle has a single masechta of the Talmud Yerushalmi - Shekalim as part of the cycle.
045         * Unlike the Bavli where the number of daf per masechta was standardized since the original <a
046         * href="http://en.wikipedia.org/wiki/Daniel_Bomberg">Bomberg Edition</a> published from 1520 - 1523, there is no
047         * uniform page length in the Yerushalmi. The early cycles had the Yerushalmi Shekalim length of 13 days following
048         * the <a href="http://en.wikipedia.org/wiki/Zhytomyr">Zhytomyr</a> Shas used by <a
049         * href="http://en.wikipedia.org/wiki/Meir_Shapiro">Rabbi Meir Shapiro</a>. With the start of the eighth Daf Yomi
050         * cycle beginning on June 24, 1975 the length of the Yerushalmi shekalim was changed from 13 to 22 daf to follow
051         * the Vilna Shas that is in common use today.
052         * 
053         * @param date
054         *            the calendar date for calculation
055         * @return the {@link Daf}.
056         * 
057         * @throws IllegalArgumentException
058         *             if the date is prior to the September 11, 1923 start date of the first Daf Yomi cycle
059         */
060        public static Daf getDafYomiBavli(Date date) {
061                /**
062                 * The number of daf per masechta. Since the number of blatt in Shekalim changed on the 8th Daf Yomi cycle
063                 * beginning on June 24, 1975
064                 */
065                int[] blattPerMasechta = { 64, 157, 105, 121, 22, 88, 56, 40, 35, 31, 32, 29, 27, 122, 112, 91, 66, 49, 90, 82,
066                                119, 119, 176, 113, 24, 49, 76, 14, 120, 110, 142, 61, 34, 34, 28, 22, 4, 10, 4, 73 };
067                Daf dafYomi = null;
068                int julianDay = getJulianDay(date);
069                int cycleNo = 0;
070                int dafNo = 0;
071                if (date.before(dafYomiStartDate)) {
072                        // TODO: should we return a null or throw an IllegalArgumentException?
073                        throw new IllegalArgumentException(date + " is prior to organized Daf Yomi Bavli cycles that started on "
074                                        + dafYomiStartDate);
075                }
076                if (date.equals(shekalimChangeDate) || date.after(shekalimChangeDate)) {
077                        cycleNo = 8 + (int) ((julianDay - shekalimJulianChangeDay) / 2711);
078                        dafNo = ((julianDay - shekalimJulianChangeDay) % 2711);
079                } else {
080                        cycleNo = 1 + (int) ((julianDay - dafYomiJulianStartDay) / 2702);
081                        dafNo = ((julianDay - dafYomiJulianStartDay) % 2702);
082                }
083
084                int total = 0;
085                int masechta = -1;
086                int blatt = 0;
087
088                /* Fix Shekalim for old cycles. */
089                if (cycleNo <= 7) {
090                        blattPerMasechta[4] = 13;
091                }
092                /* Finally find the daf. */
093                for (int j = 0; j < blattPerMasechta.length; j++) {
094                        masechta++;
095                        total = total + blattPerMasechta[j] - 1;
096                        if (dafNo < total) {
097                                blatt = 1 + blattPerMasechta[j] - (total - dafNo);
098                                /* Fiddle with the weird ones near the end. */
099                                if (masechta == 36) {
100                                        blatt += 21;
101                                } else if (masechta == 37) {
102                                        blatt += 24;
103                                } else if (masechta == 38) {
104                                        blatt += 33;
105                                }
106                                dafYomi = new Daf(masechta, blatt);
107                                break;
108                        }
109                }
110
111                return dafYomi;
112        }
113
114        /**
115         * Return the <a href="http://en.wikipedia.org/wiki/Julian_day">Julian day</a> from a Java Date.
116         * 
117         * @param date
118         *            The Java Date
119         * @return the Julian day number corresponding to the date
120         */
121        private static int getJulianDay(Date date) {
122                Calendar calendar = Calendar.getInstance();
123                calendar.setTime(date);
124                int year = calendar.get(Calendar.YEAR);
125                int month = calendar.get(Calendar.MONTH) + 1;
126                int day = calendar.get(Calendar.DAY_OF_MONTH);
127                if (month <= 2) {
128                        year -= 1;
129                        month += 12;
130                }
131                int a = year / 100;
132                int b = 2 - a + a / 4;
133                return (int) (Math.floor(365.25 * (year + 4716)) + Math.floor(30.6001 * (month + 1)) + day + b - 1524.5);
134        }
135}
