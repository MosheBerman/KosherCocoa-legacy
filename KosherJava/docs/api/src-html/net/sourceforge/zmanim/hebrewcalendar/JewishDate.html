022/**
023 * The JewishDate class allows one to maintain an instance of a Gregorian date along with the corresponding Jewish date.
024 * This class can use the standard Java Date and Calendar classes for setting it, but does not subclass these classes or
025 * use them internally to any extensive use. This class also does not have a concept of a time (which the Date class
026 * does). Please note that the calendar does not currently support dates prior to 1/1/1 Gregorian. Also keep in mind
027 * that the Gregorian calendar started on October 15, 1582, so any calculations prior to that are suspect (at least from
028 * a Gregorian perspective). While 1/1/1 Gregorian and forward are technically supported, any calculations prior to <a
029 * href="http://en.wikipedia.org/wiki/Hillel_II">Hillel II's (Hakatan's</a>) calendar (4119 in the Jewish Calendar / 359
030 * CE Julian as recorded by <a href="http://en.wikipedia.org/wiki/Hai_Gaon">Rav Hai Gaon</a>) would be just an
031 * approximation.
032 * 
033 * This open source Java code was written by <a href="http://www.facebook.com/avromf">Avrom Finkelstien</a> from his C++
034 * code. It was refactored to fit the KosherJava Zmanim API with simplification of the code, enhancements and some bug
035 * fixing.
036 * 
037 * Some of Avrom's original C++ code was translated from <a href="http://emr.cs.uiuc.edu/~reingold/calendar.C">C/C++
038 * code</a> in <a href="http://www.calendarists.com">Calendrical Calculations</a> by Nachum Dershowitz and Edward M.
039 * Reingold, Software-- Practice & Experience, vol. 20, no. 9 (September, 1990), pp. 899- 928. Any method with the mark
040 * "ND+ER" indicates that the method was taken from this source with minor modifications.
041 * 
042 * If you are looking for a class that implements a Jewish calendar version of the Calendar class, one is available from
043 * the <a href="http://site.icu-project.org/" >ICU (International Components for Unicode)</a> project, formerly part of
044 * IBM's DeveloperWorks.
045 * 
046 * @see net.sourceforge.zmanim.hebrewcalendar.JewishCalendar
047 * @see net.sourceforge.zmanim.hebrewcalendar.HebrewDateFormatter
048 * @see java.util.Date
049 * @see java.util.Calendar
050 * @author &copy; Avrom Finkelstien 2002
051 * @author &copy; Eliyahu Hershfeld 2011
052 * @version 0.2.6
053 */

323        /**
324         * Computes the Gregorian date from the absolute date. ND+ER
325         */
326        private void absDateToDate(int absDate) {
327                int year = absDate / 366; // Search forward year by year from approximate year
328                while (absDate >= gregorianDateToAbsDate(year + 1, 1, 1)) {
329                        year++;
330                }
331
332                int month = 1; // Search forward month by month from January
333                while (absDate > gregorianDateToAbsDate(year, month, getLastDayOfGregorianMonth(month, year))) {
334                        month++;
335                }
336
337                int dayOfMonth = absDate - gregorianDateToAbsDate(year, month, 1) + 1;
338                setInternalGregorianDate(year, month, dayOfMonth);
339        }

350        /**
351         * Computes the absolute date from a Gregorian date. ND+ER
352         * 
353         * @param year
354         *            the Gregorian year
355         * @param month
356         *            the Gregorian month. Unlike the Java Calendar where January has the value of 0,This expects a 1 for
357         *            January
358         * @param dayOfMonth
359         *            the day of the month (1st, 2nd, etc...)
360         * @return the absolute Gregorian day
361         */
362        private static int gregorianDateToAbsDate(int year, int month, int dayOfMonth) {
363                int absDate = dayOfMonth;
364                for (int m = month - 1; m > 0; m--) {
365                        absDate += getLastDayOfGregorianMonth(m, year); // days in prior months of the year
366                }
367                return (absDate // days this year
368                                + 365 * (year - 1) // days in previous years ignoring leap days
369                                + (year - 1) / 4 // Julian leap days before this year
370                                - (year - 1) / 100 // minus prior century years
371                + (year - 1) / 400); // plus prior years divisible by 400
372        }

451        /**
452         * Adds the 4 dechiyos for molad Tishrei. These are:
453         * <ol>
454         * <li>Lo ADU Rosh - Rosh Hashana can't fall on a Sunday, Wednesday or Friday. If the molad fell on one of these
455         * days, Rosh Hashana is delayed to the following day.</li>
456         * <li>Molad Zaken - If the molad of Tishrei falls after 12 noon, Rosh Hashana is delayed to the following day. If
457         * the following day is ADU, it will be delayed an additional day.</li>
458         * <li>GaTRaD - If on a non leap year the molad of Tishrei falls on a Tuesday (Ga) on or after 9 hours (T) and 204
459         * chalakim (TRaD) it is delayed till Thursday (one day delay, plus one day for Lo ADU Rosh)</li>
460         * <li>BeTuTaKFoT - if the year following a leap year falls on a Monday (Be) on or after 15 hours (Tu) and 589
461         * chalakim (TaKFoT) it is delayed till Tuesday</li>
462         * </ol>
463         * 
464         * @param year
465         * @param moladDay
466         * @param moladParts
467         * @return the nimber of elapsed days in the JewishCalendar adjusted for the 4 dechiyos.
468         */
469        private static int addDechiyos(int year, int moladDay, int moladParts) {
470                int roshHashanaDay = moladDay; // if no dechiyos
471                // delay Rosh Hashana for the dechiyos of the Molad - new moon 1 - Molad Zaken, 2- GaTRaD 3- BeTuTaKFoT
472                if ((moladParts >= 19440) // Dechiya of Molad Zaken - molad is >= midday (18 hours * 1080 chalakim)
473                                || (((moladDay % 7) == 2) // start Dechiya of GaTRaD - Ga = is a Tuesday
474                                                && (moladParts >= 9924) // TRaD = 9 hours, 204 parts or later (9 * 1080 + 204)
475                                && !isJewishLeapYear(year)) // of a non-leap year - end Dechiya of GaTRaD
476                                || (((moladDay % 7) == 1) // start Dechiya of BeTuTaKFoT - Be = is on a Monday
477                                                && (moladParts >= 16789) // TRaD = 15 hours, 589 parts or later (15 * 1080 + 589)
478                                && (isJewishLeapYear(year - 1)))) { // in a year following a leap year - end Dechiya of BeTuTaKFoT
479                        roshHashanaDay += 1; // Then postpone Rosh HaShanah one day
480                }
481                // start 4th Dechiya - Lo ADU Rosh - Rosh Hashana can't occur on A- sunday, D- Wednesday, U - Friday
482                if (((roshHashanaDay % 7) == 0)// If Rosh HaShanah would occur on Sunday,
483                                || ((roshHashanaDay % 7) == 3) // or Wednesday,
484                                || ((roshHashanaDay % 7) == 5)) { // or Friday - end 4th Dechiya - Lo ADU Rosh
485                        roshHashanaDay = roshHashanaDay + 1; // Then postpone it one (more) day
486                }
487                return roshHashanaDay;
488        }

490        /**
491         * Returns the number of chalakim (parts - 1080 to the hour) from the original hypothetical Molad Tohu to the year
492         * and month passed in.
493         * 
494         * @param jewishYear
495         * @param jewishMonth
496         *            the Jewish month, with the month numbers starting from Nisan. Use the JewishDate constants such as
497         *            {@link JewishDate#TISHREI}.
498         * @return the number of chalakim (parts - 1080 to the hour) from the original hypothetical Molad Tohu
499         */
500        public static long getChalakimSinceMoladTohu(int jewishYear, int jewishMonth) {
501                // Jewish lunar month = 29 days, 12 hours and 793 chalakim
502                // chalakim since Molad Tohu BeHaRaD - 1 day, 5 hours and 204 chalakim
503                int monthOfYear = getJewishMonthOfYear(jewishYear, jewishMonth);
504                int monthsElapsed = (235 * ((jewishYear - 1) / 19)) // Months in complete 19 year lunar (Metonic) cycles so far
505                                + (12 * ((jewishYear - 1) % 19)) // Regular months in this cycle
506                                + ((7 * ((jewishYear - 1) % 19) + 1) / 19) // Leap months this cycle
507                                + (monthOfYear - 1); // add elapsed months till the start of the molad of the month
508                // return chalakim prior to BeHaRaD + number of chalakim since
509                return CHALAKIM_MOLAD_TOHU + (CHALAKIM_PER_MONTH * monthsElapsed);
510        }

777         * Computes the Jewish date from the absolute date. ND+ER
778         */
779        private void absDateToJewishDate() {
780                // Approximation from below
781                jewishYear = (gregorianAbsDate + JEWISH_EPOCH) / 366;
782                // Search forward for year from the approximation
783                while (gregorianAbsDate >= jewishDateToAbsDate(jewishYear + 1, 7, 1)) {
784                        jewishYear++;
785                }
786                // Search forward for month from either Tishri or Nisan.
787                if (gregorianAbsDate < jewishDateToAbsDate(jewishYear, 1, 1)) {
788                        jewishMonth = TISHREI;// Start at Tishri
789                } else {
790                        jewishMonth = NISSAN;// Start at Nisan
791                }
792                while (gregorianAbsDate > jewishDateToAbsDate(jewishYear, jewishMonth,
793                                getDaysInJewishMonth(jewishMonth, jewishYear))) {
794                        jewishMonth++;
795                }
796                // Calculate the day by subtraction
797                jewishDay = gregorianAbsDate - jewishDateToAbsDate(jewishYear, jewishMonth, 1) + 1;
798        }

1209         * Rolls the date back by 1 day. It modifies both the Gregorian and Jewish dates accordingly. The API does not
1210         * currently offer the ability to forward more than one day t a time, or to forward by month or year. If such
1211         * manipulation is required use the {@link Calendar} class {@link Calendar#add(int, int)} or
1212         * {@link Calendar#roll(int, int)} methods in the following manner.
1213         * 
1214         * <pre>
1215         * <code>
1216         *      Calendar cal = jewishDate.getTime(); // get a java.util.Calendar representation of the JewishDate
1217         *      cal.add(Calendar.MONTH, -3); // subtract 3 Gregorian months
1218         *      jewishDate.setDate(cal); // set the updated calendar back to this class
1219         * </code>
1220         * </pre>
1221         * 
1222         * @see #back()
1223         * @see Calendar#add(int, int)
1224         * @see Calendar#roll(int, int)
1225         */
1226        public void back() {

1228                if (gregorianDayOfMonth == 1) { // if first day of month
1229                        if (gregorianMonth == 1) { // if first day of year
1230                                gregorianMonth = 12;
1231                                gregorianYear--;
1232                        } else {
1233                                gregorianMonth--;
1234                        }
1235                        // change to last day of previous month
1236                        gregorianDayOfMonth = getLastDayOfGregorianMonth(gregorianMonth, gregorianYear);
1237                } else {
1238                        gregorianDayOfMonth--;
1239                }
1240                // change Jewish date
1241                if (jewishDay == 1) { // if first day of the Jewish month
1242                        if (jewishMonth == NISSAN) {
1243                                jewishMonth = getLastMonthOfJewishYear(jewishYear);
1244                        } else if (jewishMonth == TISHREI) { // if Rosh Hashana
1245                                jewishYear--;
1246                                jewishMonth--;
1247                        } else {
1248                                jewishMonth--;
1249                        }
1250                        jewishDay = getDaysInJewishMonth(jewishMonth, jewishYear);
1251                } else {
1252                        jewishDay--;
1253                }
1254
1255                if (dayOfWeek == 1) { // if first day of week, loop back to Saturday
1256                        dayOfWeek = 7;
1257                } else {
1258                        dayOfWeek--;
1259                }
1260                gregorianAbsDate--; // change the absolute date
