053        /**
054         * @return the masechtaNumber
055         */
056        public int getMasechtaNumber() {
057                return masechtaNumber;
058        }
059
060        /**
061         * Set the masechta number in the order of teh Daf Yomi. The sequence is: Berachos, Shabbos, Eruvin, Pesachim,
062         * Shekalim, Yoma, Sukkah, Beitzah, Rosh Hashana, Taanis, Megillah, Moed Katan, Chagigah, Yevamos, Kesubos, Nedarim,
063         * Nazir, Sotah, Gitin, Kiddushin, Bava Kamma, Bava Metzia, Bava Basra, Sanhedrin, Makkos, Shevuos, Avodah Zarah,
064         * Horiyos, Zevachim, Menachos, Chullin, Bechoros, Arachin, Temurah, Kerisos, Meilah, Kinnim, Tamid, Midos and
065         * Niddah.
066         * 
067         * @param masechtaNumber
068         *            the masechtaNumber to set
069         */
070        public void setMasechtaNumber(int masechtaNumber) {
071                this.masechtaNumber = masechtaNumber;
072        }
073
074        /**
075         * Constructor that creates a Daf setting the {@link #setMasechtaNumber(int) masechta Number} and
076         * {@link #setDaf(int) daf Number}
077         * 
078         * @param masechtaNumber
079         * @param daf
080         */
081        public Daf(int masechtaNumber, int daf) {
082                this.masechtaNumber = masechtaNumber;
083                this.daf = daf;
084        }
085
086        /**
087         * Returns the daf (page number) of the Daf Yomi
088         * @return the daf (page number) of the Daf Yomi
089         */
090        public int getDaf() {
091                return daf;
092        }
093
094        /**
095         * sets the daf (page number) of the Daf Yomi
096         * @param daf daf (page number)
097         */
098        public void setDaf(int daf) {
099                this.daf = daf;
100        }
101
102        /**
103         * Returns the transliterated name of the masechta (tractate) of the Daf Yomi. The list of mashechtos is: Berachos,
104         * Shabbos, Eruvin, Pesachim, Shekalim, Yoma, Sukkah, Beitzah, Rosh Hashana, Taanis, Megillah, Moed Katan, Chagigah,
105         * Yevamos, Kesubos, Nedarim, Nazir, Sotah, Gitin, Kiddushin, Bava Kamma, Bava Metzia, Bava Basra, Sanhedrin,
106         * Makkos, Shevuos, Avodah Zarah, Horiyos, Zevachim, Menachos, Chullin, Bechoros, Arachin, Temurah, Kerisos, Meilah,
107         * Kinnim, Tamid, Midos and Niddah.
108         * 
109         * @return the transliterated name of the masechta (tractate) of the Daf Yomi such as Berachos.
110         */
111        public String getMasechtaTransliterated() {
112                return masechtosBavliTransliterated[daf];
113        }
114
115        /**
116         * Returns the masechta (tractate) of the Daf Yomi in Hebrew, It will return
117         * &#x05D1;&#x05E8;&#x05DB;&#x05D5;&#x05EA; for Berachos.
118         * 
119         * @return the masechta (tractate) of the Daf Yomi in Hebrew, It will return
120         *         &#x05D1;&#x05E8;&#x05DB;&#x05D5;&#x05EA; for Berachos.
121         */
122        public String getMasechta() {
123                return masechtosBavli[daf];
124        }
