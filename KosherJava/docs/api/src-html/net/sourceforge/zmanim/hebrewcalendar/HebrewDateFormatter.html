170        /**
171         * Formats the Yom Tov (holiday) in Hebrew or transliterated Latin characters.
172         * 
173         * @param jewishCalendar
174         * @return the formatted holiday or an empty String if the day is not a holiday.
175         * @see #isHebrewFormat()
176         */
177        public String formatYomTov(JewishCalendar jewishCalendar) {
178                int index = jewishCalendar.getYomTovIndex();
179                if (index == JewishCalendar.CHANUKAH) {
180                        int dayOfChanukah = jewishCalendar.getDayOfChanukah();
181                        return hebrewFormat ? (formatHebrewNumber(dayOfChanukah) + " " + hebrewHolidays[index])
182                                        : (transliteratedHolidays[index] + dayOfChanukah);
183                }
184                return index == -1 ? "" : hebrewFormat ? hebrewHolidays[index] : transliteratedHolidays[index];
185        }
186
187        public String formatRoshChodesh(JewishCalendar jewishCalendar) {
188                if (!jewishCalendar.isRoshChodesh()) {
189                        return "";
190                }
191                String formattedRoshChodesh = "";
192                int month = jewishCalendar.getJewishMonth();
193                if (jewishCalendar.getJewishDayOfMonth() == 30) {
194                        if (month < 12 || (month == 12 && jewishCalendar.isJewishLeapYear())) {
195                                month++;
196                        } else { // roll to Nissan
197                                month = 1;
198                        }
199                }
200                formattedRoshChodesh = hebrewFormat ? hebrewHolidays[JewishCalendar.ROSH_CHODESH]
201                                : transliteratedHolidays[JewishCalendar.ROSH_CHODESH];
202                formattedRoshChodesh += " " + formatMonth(jewishCalendar.getJewishYear(), month);
203                return formattedRoshChodesh;
204        }

538        private String formatMonth(int year, int month){
539                if (isHebrewFormat()) {
540                        if (JewishDate.isJewishLeapYear(year) && month == JewishDate.ADAR) {
541                                return hebrewMonths[13] + (useGershGershayim ? GERESH : ""); // return Adar I, not Adar in a leap year
542                        } else if (JewishDate.isJewishLeapYear(year) && month == JewishDate.ADAR_II) {
543                                return hebrewMonths[12] + (useGershGershayim ? GERESH : "");
544                        } else {
545                                return hebrewMonths[month - 1];
546                        }
547                } else {
548                        if (JewishDate.isJewishLeapYear(year) && month == JewishDate.ADAR) {
549                                return transliteratedMonths[13]; // return Adar I, not Adar in a leap year
550                        } else {
551                                return transliteratedMonths[month - 1];
552                        }
553                }
554        }

608        /**
609         * Returns the kviah in the traditional 3 letter Hebrew format where the first letter represents the day of week of
610         * Rosh Hashana, the second letter represents the lengths of Cheshvan and Kislev ({@link JewishDate#SHELAIMIM
611         * Shelaimim} , {@link JewishDate#KESIDRAN Kesidran} or {@link JewishDate#CHASERIM Chaserim}) and the 3rd letter
612         * represents the day of week of Pesach. For example 5729 (1969) would return &#x5D1;&#x5E9;&#x5D4; (Rosh Hashana on
613         * Monday, Shelaimim, and Pesach on Thursday), while 5771 (2011) would return &#x5D4;&#x5E9;&#x5D2; (Rosh Hashana on
614         * Thursday, Shelaimim, and Pesach on Tuesday).
615         * 
616         * @param jewishYear
617         *            the Jewish year
618         * @return the Hebrew String such as &#x5D1;&#x5E9;&#x5D4; for 5729 (1969) and &#x5D4;&#x5E9;&#x5D2; for 5771
619         *         (2011).
620         */
621        public String getFormattedKviah(int jewishYear) {
622                JewishDate jewishDate = new JewishDate(jewishYear, JewishDate.TISHREI, 1); // set date to Rosh Hashana
623                int kviah = JewishCalendar.getCheshvanKislevKviah(jewishYear);
624                int roshHashanaDayOfweek = jewishDate.getDayOfWeek();
625                String returnValue = formatHebrewNumber(roshHashanaDayOfweek);
626                returnValue += (kviah == JewishCalendar.CHASERIM ? "\u05D7" : kviah == JewishCalendar.SHELAIMIM ? "\u05E9"
627                                : "\u05DB");
628                jewishDate.setJewishDate(jewishYear, JewishDate.NISSAN, 15); // set to Pesach of the given year
629                int pesachDayOfweek = jewishDate.getDayOfWeek();
630                returnValue += formatHebrewNumber(pesachDayOfweek);
631                returnValue = returnValue.replaceAll(GERESH, "");// geresh is never used in the kviah format

633                // for efficiency we can avoid the expensive recalculation of the pesach day of week by adding 1 day to Rosh
634                // Hashana for a 353 day year, 2 for a 354 day year, 3 for a 355 or 383 day year, 4 for a 384 day year and 5 for
635                // a 385 day year
636                return returnValue;

647        /**
648         * Returns a Hebrew formatted string of a number. The method can calculate from 0 - 9999.
649         * <ul>
650         * <li>Single digit numbers such as 3, 30 and 100 will be returned with a &#x5F3; (<a
651         * href="http://en.wikipedia.org/wiki/Geresh">Geresh</a>) appended as at the end. For example &#x5D2;&#x5F3;,
652         * &#x5DC;&#x5F3; and &#x5E7;&#x5F3;</li>
653         * <li>multi digit numbers such as 21 and 769 will be returned with a &#x5F4; (<a
654         * href="http://en.wikipedia.org/wiki/Gershayim">Gershayim</a>) between the second to last and last letters. For
655         * example &#x5DB;&#x5F4;&#x5D0;, &#x5EA;&#x5E9;&#x5DB;&#x5F4;&#x5D8;</li>
656         * <li>15 and 16 will be returned as &#x5D8;&#x5F4;&#x5D5; and &#x5D8;&#x5F4;&#x5D6;</li>
657         * <li>Single digit numbers (years assumed) such as 6000 (%1000=0) will be returned as &#x5D5;&#x5F3;
658         * &#x5D0;&#x5DC;&#x5E4;&#x5D9;&#x5DD;</li>
659         * <li>0 will return &#x5D0;&#x5E4;&#x05E1;</li>
660         * </ul>
661         * 
662         * @param number
663         *            the number to be formatted. It will trow an IllegalArgumentException if the number is < 0 or > 9999.
664         * @return the Hebrew formatted number such as &#x5EA;&#x5E9;&#x5DB;&#x5F4;&#x5D8;
665         * 
666         */
667        public String formatHebrewNumber(int number) {
668                if (number < 0) {
669                        throw new IllegalArgumentException("negative numbers can't be formatted");
670                } else if (number > 9999) {
671                        throw new IllegalArgumentException("numbers > 9999 can't be formatted");
672                }
673
674                String ALAFIM = "\u05D0\u05DC\u05E4\u05D9\u05DD";
675                String EFES = "\u05D0\u05E4\u05E1";
676
677                String[] jHundreds = new String[] { "", "\u05E7", "\u05E8", "\u05E9", "\u05EA", "\u05EA\u05E7", "\u05EA\u05E8",
678                                "\u05EA\u05E9", "\u05EA\u05EA", "\u05EA\u05EA\u05E7" };
679                String[] jTens = new String[] { "", "\u05D9", "\u05DB", "\u05DC", "\u05DE", "\u05E0", "\u05E1", "\u05E2",
680                                "\u05E4", "\u05E6" };
681                String[] jTenEnds = new String[] { "", "\u05D9", "\u05DA", "\u05DC", "\u05DD", "\u05DF", "\u05E1", "\u05E2",
682                                "\u05E3", "\u05E5" };
683                String[] tavTaz = new String[] { "\u05D8\u05D5", "\u05D8\u05D6" };
684                String[] jOnes = new String[] { "", "\u05D0", "\u05D1", "\u05D2", "\u05D3", "\u05D4", "\u05D5", "\u05D6",
685                                "\u05D7", "\u05D8" };
686
687                if (number == 0) { // do we realy need this? Should it be applicable to a date?
688                        return EFES;
689                }
690                int shortNumber = number % 1000; // discard thousands
691                // next check for all possible single Hebrew digit years
692                boolean singleDigitNumber = (shortNumber < 11 || (shortNumber < 100 && shortNumber % 10 == 0) || (shortNumber <= 400 && shortNumber % 100 == 0));
693                int thousands = number / 1000; // get # thousands
694                StringBuffer returnBuffer = new StringBuffer();
695                // append thousands to String
696                if (number % 1000 == 0) { // in year is 5000, 4000 etc
697                        returnBuffer.append(jOnes[thousands]);
698                        if (isUseGershGershayim()) {
699                                returnBuffer.append(GERESH);
700                        }
701                        returnBuffer.append(" ");
702                        returnBuffer.append(ALAFIM); // add # of thousands plus word thousand (overide alafim boolean)
703                        return returnBuffer.toString();
704                } else if (useLonghebrewYears && number >= 1000) { // if alafim boolean display thousands
705                        returnBuffer.append(jOnes[thousands]);
706                        if (isUseGershGershayim()) {
707                                returnBuffer.append(GERESH); // append thousands quote
708                        }
709                        returnBuffer.append(" ");
710                }
711                number = number % 1000; // remove 1000s
712                int hundreds = number / 100; // # of hundreds
713                returnBuffer.append(jHundreds[hundreds]); // add hundreds to String
714                number = number % 100; // remove 100s
715                if (number == 15) { // special case 15
716                        returnBuffer.append(tavTaz[0]);
717                } else if (number == 16) { // special case 16
718                        returnBuffer.append(tavTaz[1]);
719                } else {
720                        int tens = number / 10;
721                        if (number % 10 == 0) { // if evenly divisable by 10
722                                if (singleDigitNumber == false) {
723                                        returnBuffer.append(jTenEnds[tens]); // end letters so years like 5750 will end with an end nun
724                                } else {
725                                        returnBuffer.append(jTens[tens]); // standard letters so years like 5050 will end with a regular nun
726                                }
727                        } else {
728                                returnBuffer.append(jTens[tens]);
729                                number = number % 10;
730                                returnBuffer.append(jOnes[number]);
731                        }
732                }
733                if (isUseGershGershayim()) {
734                        if (singleDigitNumber == true) {
735                                returnBuffer.append(GERESH); // append single quote
736                        } else { // append double quote before last digit
737                                returnBuffer.insert(returnBuffer.length() - 1, GERSHAYIM);
738                        }
739                }
740                return returnBuffer.toString();
741        }
